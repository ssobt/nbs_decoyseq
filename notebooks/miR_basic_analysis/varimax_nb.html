zero_below_thresh <- function(df, cutoffs){
    df = as.matrix(df)
    for (i in 1:ncol(df)){
    df[,i][abs(df[,i]) < cutoffs[i]] = 0
    colnames(df)[i] = paste0('RC', i)
    }
    return(df)
}
